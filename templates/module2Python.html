                            #Se crea la clase del nodo
                            class Node(object):
                                #Se inicializan los valores
                                def __init__(self, object=None, next=None, before=None):
                                    self.object = object
                                    self.next = next
                                    self.before = before

                                #Muestra el valor del objeto
                                def getObject(self):
                                    return self.object

                                #Muestra el valor del siguiente objeto
                                def getNext(self):
                                    return self.next

                                #Muestra el valor del siguiente objeto
                                def setNext(self, next):
                                    self.next = next

                                #Muestra el valor del anterior objeto
                                def getBefore(self):
                                    return self.before

                                #Modifica el valor del anterior objeto
                                def setBefore(self, before):
                                    self.before = before


                            #Se crea la clase de la lista
                            class List(object):
                                def __init__(self, head=None):
                                    self.head = head
                                    self.tail = head
                                    self.size = 0

                                def __iter__(self):
                                    inode = self.head
                                    for i in range(self.size):
                                        yield inode.object
                                        inode = inode.next

                                #Retorna un true si la lista esta vacia y no si tiene objectos
                                def isEmpty(self):
                                    return self.head is None

                                #Retorna la cantidad de objetos
                                def getSize(self):
                                    inode = self.head
                                    acum = 0
                                    while inode:
                                        acum += 1
                                        inode = inode.getNext()
                                    return acum

                                #Se borran todos los elementos de la lista
                                def clear(self):
                                    self.head = None
                                    self.tail = None
                                    self.size = 0

                                #Se obtiene el primer valor de la lista
                                def getHead(self):
                                    if self.isEmpty():
                                        raise IndexError("la lista esta vacia")
                                    return self.head.object

                                #Se obtiene el ultimo valor de la lista
                                def getTail(self):
                                    if self.isEmpty():
                                        raise IndexError("la lista esta vacia")
                                    return self.tail.object

                                #Se busca un objeto en la lista
                                def search(self, object):
                                    if self.isEmpty():
                                        raise IndexError("la lista esta vacia")
                                    inode = self.head
                                    for i in range(self.size):
                                        if inode.object == object:
                                            return i
                                        inode = inode.next
                                    return -1

                                #Se añade un objeto a la lista
                                def add(self, object):
                                    if self.size == 0:
                                        self.head = Node(object)
                                        self.tail = self.head
                                    else:
                                        previous = self.tail
                                        following = Node(object, before=previous)
                                        previous.next = following
                                        self.tail = following
                                    self.size += 1

                                #Se añade un valor a la lista
                                def insert(self, index, object):
                                    if self.isEmpty():
                                        raise IndexError("la lista esta vacia")
                                    if self.size == 1:
                                        previous = self.head
                                        following = Node(object)
                                        following.next = previous
                                        self.head = following
                                        self.tail = previous
                                        self.tail.before = self.head
                                    elif index == 0:
                                        previous = self.head
                                        following = Node(object, previous)
                                        previous.before = following
                                        self.head = following
                                    elif index == self.size - 1:
                                        inode = Node(object, self.tail, self.tail.before)
                                        self.tail.before.next = inode
                                        self.tail.before = inode
                                    else:
                                        inode = self.head
                                        for i in range(self.size):
                                            if index == i:
                                                inode = Node(object, inode, inode.before)
                                                inode.before.next = inode
                                                inode.before = inode
                                            inode = inode.next
                                    self.size += 1

                                #Se inserta un objeto a la primera posicion
                                def insertHead(self, object):
                                    self.insert(0, object)

                                #Se inserta un objeto a la ultima posicion
                                def insertTail(self, object):
                                    self.insert(self.size - 1, object)

                                #Se elimina un valor de la lista
                                def remove(self, index):
                                    if self.isEmpty():
                                        raise IndexError("la lista esta vacia")
                                    if self.size == 1:
                                        self.clear()
                                        return
                                    elif index == 0:
                                        self.head.next.before = None
                                        self.head = self.head.next
                                    elif index == self.size - 1:
                                        self.tail.before.next = None
                                        self.tail = self.tail.before
                                    else:
                                        inode = self.head
                                        for i in range(self.size):
                                            if index == i:
                                                inode.next.before = inode.before
                                                inode.before.next = inode.next
                                            inode = inode.next
                                    self.size -= 1

                                #Metodo si contiene un objeto
                                def contains(self, object):
                                    return self.search(object) != -1

                                #Se crea una lista definiendo el inicio y el fin
                                def subList(self, From, To):
                                    if self.isEmpty():
                                        raise IndexError("la lista esta vacia")
                                    if To > self.size:
                                        To = self.size
                                    subList = List()
                                    var = False
                                    inode = self.head
                                    for i in range(self.size):
                                        if i == From:
                                            subList.add(inode.object)
                                            var = True
                                            if From == To:
                                                break
                                        elif var and i < To:
                                            subList.add(inode.object)
                                        inode = inode.next
                                    return subList

                                #Transforma la lista a un array
                                def toArray(self):
                                    if self.isEmpty():
                                        raise IndexError("la lista esta vacia")
                                    else:
                                        array = [self.size]
                                        inode = self.head
                                        for i in range(self.size):
                                            array[i] = inode.object
                                            inode = inode.next
                                        return array

                                #Se muestran todos los valores de la lista
                                def toString(self):
                                    return list(i for i in self)

                                    old_tail.next = new_tail

                            def insert(self, index: int, object: object):
                                if self.isEmpty():
                                    raise IndexError("la lista esta vacia")
                                inode = ListNode.ListNode(object)
                                if index == 0:
                                    self.head.before = inode
                                    inode.next = self.head
                                    self.head = inode
                                elif index == self.size - 1:
                                    self.tail.before.next = inode
                                    inode.before = self.tail.before
                                    self.tail.before = inode
                                    inode.next = self.tail
                                else:
                                    for i, node in enumerate(self):
                                        if i == index:
                                            inode.next = node
                                            inode.before = node.before
                                            node.before.next = inode
                                            node.before = inode
                                            break
                                self.size += 1

                                if self.isEmpty():
                                    raise IndexError("la lista esta vacia")
                                inode = self.head
                                found = False
                                if self.size == 1 and self.head == node:
                                    self.head = None
                                    self.tail = None
                                    found = True
                                else:
                                    for i in range(self.size):
                                        if inode == node:
                                            if i == 0:
                                                self.head = self.head.next
                                                self.head.before = None
                                            elif i == self.size - 1:
                                                self.tail = self.tail.before
                                                self.tail.next = None
                                            else:
                                                inode.next.before = inode.before
                                                inode.before.next = inode.next
                                            found = True
                                            break
                                        inode = inode.next
                                if not found:
                                    raise ValueError("Node not in the list")
                                self.size -= 1
